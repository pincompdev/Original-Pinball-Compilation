	IF temp_int

	PAUSE_TIMER score_blink_timer
	GET_ARRAY_ELEMENT score player temp_int
	SET_INTEGER temp_int_2 replay_score
	LESS temp_int_2 temp_int temp_int_3
	WHILE temp_int_3
		ADD temp_int_2 100000 temp_int_2
		LESS temp_int_2 temp_int temp_int_3
	WEND
	ADD temp_int score_pending temp_int_3
	LESS temp_int_3 temp_int_2 temp_int_3
	NOT temp_int_3 temp_int_3

		PLAY_SOUND knocker 64
